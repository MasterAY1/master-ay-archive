WebGL Product Viewer

Interactive 3D models using React Three Fiber, allowing 360-degree product exploration directly in the browser.

components/ProductViewer.jsx 
import { useRef, useEffect } from 'react';
import { useFrame } from '@react-three/fiber';
import { useGLTF } from '@react-three/drei';
import gsap from 'gsap';

// Core WebGL Component for Product Showcase
export default function ProductViewer({ url }) {
  const modelRef = useRef();
  const { scene } = useGLTF(url);

  // Fluid entrance animation
  useEffect(() => {
    gsap.from(modelRef.current.scale, {
      x: 0, y: 0, z: 0,
      duration: 1.5,
      ease: "expo.out"
    });
  }, []);

  return (
    <primitive 
      object={scene} 
      ref={modelRef} 
    />
  );
}
